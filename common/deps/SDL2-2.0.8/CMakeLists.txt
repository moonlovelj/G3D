set(SDL_ROOT_DIR ${CMAKE_CURRENT_SOURCE_DIR} CACHE PATH "" FORCE)
set(SDL_HEADER_DIR ${SDL_ROOT_DIR}/include)
file(GLOB SDL_DLLS ${SDL_ROOT_DIR}/lib/*.dll)

include_directories(${SDL_HEADER_DIR})

add_library(SDL2 SHARED IMPORTED)

set(SDL_LIBRARY_DIRS "${SDL_ROOT_DIR}/lib" CACHE PATH "" FORCE)
set(SDL_LIBRARIES SDL2 CACHE PATH "" FORCE)
set(SDL_INCLUDE_DIRS ${SDL_HEADER_DIR} CACHE PATH "" FORCE)
set(SDL_RUNTIMELIBS ${SDL_DLLS} CACHE PATH "" FORCE)
